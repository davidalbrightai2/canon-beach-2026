Set up your trip planner in 3 steps

1 Create Your Google Sheet

Create a new Google Sheet with these tabs:

  • activities — columns: id, day, title, description, link, suggestedBy, upvotes, downvotes, timestamp
  • meals — columns: id, day, mealType, claimedBy, description, isDYOT, timestamp
  • config — columns: key, value

Add headers or let the script do it for you.

2 Add the Apps Script

In your Google Sheet: Extensions → Apps Script

Replace the code and deploy as Web app (Execute as: Me, Access: Anyone):

// Cannon Beach 2026 - Google Apps Script Backend function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const action = e.parameter.action || 'getData'; try { let result; switch(action) { case 'getData': result = getData(); break; case 'claimMeal': result = claimMeal(e.parameter); break; case 'unclaimMeal': result = unclaimMeal(e.parameter); break; case 'suggestActivity': result = suggestActivity(e.parameter); break; case 'voteActivity': result = voteActivity(e.parameter); break; default: result = {success: false, error: 'Unknown action'}; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService .createTextOutput(JSON.stringify({success: false, error: error.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function getData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // Get activities const activitiesSheet = getOrCreateSheet(ss, 'activities', ['id', 'day', 'title', 'description', 'link', 'suggestedBy', 'upvotes', 'downvotes', 'timestamp']); const activitiesData = activitiesSheet.getDataRange().getValues(); const activities = dataToObjects(activitiesData); // Get meals const mealsSheet = getOrCreateSheet(ss, 'meals', ['id', 'day', 'mealType', 'claimedBy', 'description', 'isDYOT', 'timestamp']); const mealsData = mealsSheet.getDataRange().getValues(); let meals = dataToObjects(mealsData); // If meals are empty, seed them if (meals.length === 0) { seedMeals(mealsSheet); meals = dataToObjects(mealsSheet.getDataRange().getValues()); } return { success: true, activities: activities, meals: meals }; } function claimMeal(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('meals'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const idCol = headers.indexOf('id'); const claimedByCol = headers.indexOf('claimedBy'); const descCol = headers.indexOf('description'); const isDYOTCol = headers.indexOf('isDYOT'); const timestampCol = headers.indexOf('timestamp'); for (let i = 1; i < data.length; i++) { if (data[i][idCol] == params.id) { sheet.getRange(i + 1, claimedByCol + 1).setValue(params.claimedBy || ''); sheet.getRange(i + 1, descCol + 1).setValue(params.description || ''); sheet.getRange(i + 1, isDYOTCol + 1).setValue(params.isDYOT === 'true' ? 'TRUE' : 'FALSE'); sheet.getRange(i + 1, timestampCol + 1).setValue(new Date()); return {success: true}; } } return {success: false, error: 'Meal not found'}; } function unclaimMeal(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('meals'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const idCol = headers.indexOf('id'); const claimedByCol = headers.indexOf('claimedBy'); const descCol = headers.indexOf('description'); const isDYOTCol = headers.indexOf('isDYOT'); const timestampCol = headers.indexOf('timestamp'); for (let i = 1; i < data.length; i++) { if (data[i][idCol] == params.id) { sheet.getRange(i + 1, claimedByCol + 1).setValue(''); sheet.getRange(i + 1, descCol + 1).setValue(''); sheet.getRange(i + 1, isDYOTCol + 1).setValue('FALSE'); sheet.getRange(i + 1, timestampCol + 1).setValue(new Date()); return {success: true}; } } return {success: false, error: 'Meal not found'}; } function suggestActivity(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('activities'); const id = 'act_' + new Date().getTime(); const row = [ id, params.day, params.title, params.description || '', params.link || '', params.suggestedBy, '', '', new Date() ]; sheet.appendRow(row); return {success: true, id: id}; } function voteActivity(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('activities'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const idCol = headers.indexOf('id'); const upvotesCol = headers.indexOf('upvotes'); const downvotesCol = headers.indexOf('downvotes'); for (let i = 1; i < data.length; i++) { if (data[i][idCol] == params.id) { let upvotes = data[i][upvotesCol] ? data[i][upvotesCol].toString().split(',').map(p => p.trim()).filter(p => p) : []; let downvotes = data[i][downvotesCol] ? data[i][downvotesCol].toString().split(',').map(p => p.trim()).filter(p => p) : []; // Remove from both lists first upvotes = upvotes.filter(u => u !== params.name); downvotes = downvotes.filter(d => d !== params.name); // Add to the appropriate list if (params.vote === 'up') { upvotes.push(params.name); } else if (params.vote === 'down') { downvotes.push(params.name); } // If vote is 'remove', just leave it out of both lists sheet.getRange(i + 1, upvotesCol + 1).setValue(upvotes.join(', ')); sheet.getRange(i + 1, downvotesCol + 1).setValue(downvotes.join(', ')); return {success: true}; } } return {success: false, error: 'Activity not found'}; } function getOrCreateSheet(ss, name, headers) { let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length).setFontWeight('bold'); } return sheet; } function seedMeals(sheet) { const mealTypes = ['Breakfast', 'Lunch', 'Dinner']; for (let day = 1; day <= 5; day++) { for (let mealType of mealTypes) { const id = `meal_${day}_${mealType.toLowerCase()}`; sheet.appendRow([id, day, mealType, '', '', 'FALSE', new Date()]); } } } function dataToObjects(data) { if (data.length < 2) return []; const headers = data[0]; const rows = data.slice(1); return rows.map(row => { const obj = {}; headers.forEach((header, i) => { obj[header] = row[i]; }); return obj; }).filter(obj => obj.id); }
